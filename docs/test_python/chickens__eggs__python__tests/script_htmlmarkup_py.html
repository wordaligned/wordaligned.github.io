"""Html markup functions.

This module provides a number of HTML markup utilities.
"""

def preformat(str):
    """Return a preformatted version of the string.
    
    >>> preformat('preformat me')
    '<pre>preformat me</pre>'
    """
    return '<pre>' + str + '</pre>'


def italics(str):
    """Return an italicised version of the string.
    
    >>> italics('put me in italics')
    '<i>put me in italics</i>'
    """
    return '<i>' + str + '</i>'


def bold(str):
    """Return a bold version of the string.
    
    >>> bold('make me bold')
    '<b>make me bold</b>'
    """
    return '<b>' + str + '</b>'


def colour(str, colour):
    """Return a coloured version of the string.
    
    >>> colour('very important!', 
    ...        'red')
    '<font color="red">very important!</font>'
    """
    return ('<font color="' + colour + '">'
            + str +
            '</font>')


def cookForHtml(src):
    """Return the input data cooked for html.
    
    >>> cookForHtml('a > b && a < c')
    'a &gt; b &amp;&amp; a &lt; c'
    
    >>> cookForHtml('&amp;')
    '&amp;amp;'
    """
    import cgi
    return cgi.escape(src)


def _testDocstrings():
    import doctest, htmlmarkup
    return doctest.testmod(htmlmarkup)

if __name__ == "__main__":
    _testDocstrings()
